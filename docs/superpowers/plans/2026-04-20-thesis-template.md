# Thesis Template Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a complete Claude Code master thesis template repository that a cohort of students can clone and immediately start using, bootstrapped by a guided `/interview` skill.

**Architecture:** Three-layer template — a `.claude/` intelligence layer (skills, agents, hooks, settings), a `thesis/` writing layer (Markdown sections), and a `code/` + `data/` analysis layer (Python). Students never touch the intelligence layer; `/interview` rewrites `CLAUDE.md` to bake in their context. Python hooks replace the existing bash hook for cross-platform compatibility.

**Tech Stack:** Python 3.8+ (stdlib for setup/hooks, pandas/matplotlib/jupyter for analysis), Pandoc + XeLaTeX (PDF compilation), Claude Code skills/agents (Markdown prompts), Semantic Scholar public API (reference validation, no key required).

#!/usr/bin/env python3
"""
PostToolUse hook: appends a log entry to .claude/logs/activity.log
Receives JSON on stdin with hook event data.
"""
import json
import os
import sys
from datetime import datetime

def main():
    try:
        data = json.load(sys.stdin)
    except (json.JSONDecodeError, EOFError):
        data = {}

    tool_name = data.get("tool_name", "unknown")
    timestamp = datetime.now().isoformat(timespec="seconds")

    log_dir = os.path.join(".claude", "logs")
    os.makedirs(log_dir, exist_ok=True)
    log_path = os.path.join(log_dir, "activity.log")

    with open(log_path, "a", encoding="utf-8") as f:
        f.write(f"{timestamp}: {tool_name}\n")

if __name__ == "__main__":
    main()

#!/usr/bin/env python3
"""
PreToolUse hook: blocks Edit and Write tool calls targeting data/raw/.
Receives JSON on stdin. Exits non-zero to block; exits 0 to allow.
"""
import json
import os
import sys


WRITE_TOOLS = {"Edit", "Write", "NotebookEdit"}
RAW_DATA_PREFIX = os.path.join("data", "raw")


def is_raw_data_path(file_path: str) -> bool:
    # Normalise separators so forward/back slash both work
    normalised = file_path.replace("\\", "/")
    return normalised.startswith("data/raw/") or normalised == "data/raw"


def main():
    try:
        data = json.load(sys.stdin)
    except (json.JSONDecodeError, EOFError):
        sys.exit(0)

    tool_name = data.get("tool_name", "")
    tool_input = data.get("tool_input", {})

    if tool_name not in WRITE_TOOLS:
        sys.exit(0)

    file_path = tool_input.get("file_path", "")
    if is_raw_data_path(file_path):
        print(
            f"BLOCKED: Claude cannot modify files in data/raw/. "
            f"Raw data is read-only. Copy to data/clean/ before editing.",
            file=sys.stderr
        )
        sys.exit(1)

    sys.exit(0)


if __name__ == "__main__":
    main()
```

#!/usr/bin/env python3
"""
PostToolUse hook: sends a desktop notification when a long-running
Bash command (pandoc compile or validate-references) completes.
Silently skips if plyer is not installed.
"""
import json
import sys


NOTIFY_KEYWORDS = ["pandoc", "validate-references", "validate_references"]


def should_notify(data: dict) -> bool:
    if data.get("tool_name") != "Bash":
        return False
    tool_input = data.get("tool_input", {})
    command = tool_input.get("command", "")
    return any(kw in command for kw in NOTIFY_KEYWORDS)


def main():
    try:
        data = json.load(sys.stdin)
    except (json.JSONDecodeError, EOFError):
        sys.exit(0)

    if not should_notify(data):
        sys.exit(0)

    try:
        from plyer import notification
        tool_input = data.get("tool_input", {})
        command = tool_input.get("command", "command")
        notification.notify(
            title="Thesis Copilot",
            message=f"Done: {command[:60]}",
            timeout=5,
        )
    except Exception:
        # plyer not installed or display not available — skip silently
        pass

    sys.exit(0)


if __name__ == "__main__":
    main()
```

- title (from the `title = {...}` field)

- **Suspicious** — year is impossible (e.g. future year), author name not found anywhere, title is generic or implausibly short
